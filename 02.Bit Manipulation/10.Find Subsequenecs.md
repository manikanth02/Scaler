string Solution::solve(string A, string B) {
    int ASize = A.size();
    int BSize = B.size();
    int i = 0;
    int j = 0;

    while(i < ASize && j < BSize){
        if(A[i] == B[j]){
            i++;
            j++;
        }else{
            j++;
        }
    }
    if(i == ASize){
        return "YES";
    }
    return "NO";
}
