string Solution::solve(vector<int> &A) {

int count = 0;
int n = A.size();
for(int i = 0; i < n; i++){
    if(A[i] % 2 != 0){
        count++;
    }
}
if(count % 2 == 0)return "Yes";
return "No";
}
