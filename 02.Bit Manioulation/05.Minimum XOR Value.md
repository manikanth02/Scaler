int Solution::findMinXor(vector<int> &A) {
    int n = A.size();
    sort(A.begin(),A.end());
    int minimumXORValue = INT_MAX;
    for(int i = 0; i < n - 1; i++){
        int XORValue = A[i]^A[i + 1];
        minimumXORValue = min(minimumXORValue,XORValue);
    }
    return minimumXORValue;
}
