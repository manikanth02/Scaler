int Solution::uniquePathsWithObstacles(vector<vector<int> > &A) {

    // Tabulation 

    int row = A.size();
    int col = A[0].size();
    
    vector<vector<int>>dp(row + 1,vector<int>(col + 1,0));
     dp[0][1] = 1;

    for(int i = 1; i <= row; i++){
        for(int j = 1; j <= col; j++){
            if(A[ i - 1][j - 1] == 0){
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
            }
        }
    }
    return dp[row][col];
}
