int minSqr(vector<int>&dp,int n){
    if(n == 0)return 0;
    if(dp[n] == INT_MAX){
        for(int j = 1; j <= n/j; j++){
            dp[n] = min(dp[n],minSqr(dp,n - j*j) + 1);
        }
    }
    return dp[n];
}
int Solution::countMinSquares(int n) {

    vector<int>dp(n + 1,INT_MAX);
    return minSqr(dp,n);
}

// T.C::O(n)
