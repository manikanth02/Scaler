int solve( vector<int>&dp,vector<int>&A,int i){
    if(i < 0){
        return 0; 
    }
    if(dp[i] == -1){
        dp[i] = max(solve(dp, A , i - 1) , solve(dp,A,i - 2) + A[i]);
    }
    return dp[i];
}
