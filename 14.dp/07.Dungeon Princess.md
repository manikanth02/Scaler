int row;
int col;
int solve(vector<vector<int>>& dp, vector<vector<int>> & A, int i,int j){
    if(i == row - 1 && j == col - 1){
        return max(1,1 - A[i][j]);

        // suppose last cell is -7 then energy required to get out is H + -7 = 1 => H = 8
        // if last cell is 2 then energy required is H => H + 2 = 1 => H => -1 but it is not possible
        // so we take energy required is 1 and we do maximum
    }

    if(i >= row || j >= col){
        return INT_MAX;
    }

    if(dp[i][j] == -1){
        int a = solve(dp,A,i + 1,j);
        int b = solve(dp,A,i,j+1);
        dp[i][j] = max(1,min(a,b) - A[i][j]);
    }

    return dp[i][j];
}
int Solution::calculateMinimumHP(vector<vector<int> > &A) {
    row = A.size();
    col = A[0].size();
    vector<vector<int>>dp(row,vector<int>(col,-1));
    solve(dp,A,0,0);
    return dp[0][0];

}
