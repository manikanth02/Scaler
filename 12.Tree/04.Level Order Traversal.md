vector<vector<int> > Solution::solve(TreeNode* root) {
    if(root == NULL){
        return {};
    }

    vector<vector<int>>ans;
    
    queue<TreeNode*>q;
    q.push(root);
    while(q.size() > 0){
        int size = q.size();
        vector<int>level(size);

        for(int i = 0; i < size; i++){
            TreeNode* top = q.front();
            q.pop();
            level[i] = top->val;
            if(top->left){
                q.push(top->left);
            }
            if(top->right){
                q.push(top->right);
            }
        }
        ans.push_back(level);
    }

    return ans;
}
