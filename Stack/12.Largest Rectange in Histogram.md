vector<int>LeftSmallerIndex(vector<int>& A){
    int size = A.size();
    vector<int>left(size,-1);
    stack<int>st;
    for(int i = 0; i < size; i++){
        while(st.size() > 0 && A[st.top()] >= A[i]){
            st.pop();
        }
        if(st.size() > 0){
            left[i] = st.top();
        }
        st.push(i);
    }
    return left;
}

vector<int>rightSmallerIndex(vector<int>& A){
    int size = A.size();
    vector<int>right(size,size);
    stack<int>st;
    for(int i = size - 1; i >= 0; i--){
        while(st.size() > 0 && A[st.top()] >= A[i]){
            st.pop();
        }

        if(st.size() > 0){
            right[i] = st.top();
        }
        st.push(i);
    }
    return right;
}
int Solution::largestRectangleArea(vector<int> &A) {
    vector<int>left = LeftSmallerIndex(A);
    vector<int>right = rightSmallerIndex(A);
    int ans = INT_MIN;
    for(int i = 0; i < A.size() ; i++){
        int length = right[i] - left[i] - 1;
        int temp = A[i]*length;
        ans = max(ans,temp);
    }
    return ans;
}
