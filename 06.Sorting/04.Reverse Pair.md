void merge(vector<int>& A,int s,int m, int e,int& ans){
    int p1 = s;
    int p2 = m + 1;
    int p3 = 0;
    vector<int>temp(e - s + 1,0);

        // logice for above qn
        int j = m + 1;
        for(int i = s; i <= m;i++){
            while(j <= e && A[i] > 2LL * A[j]){
                j++;
            }
            ans += j - (m + 1);
        }




    while(p1 <= m && p2 <= e){
        if(A[p1] <= A[p2]){
            temp[p3++] = A[p1++];
        }else{
            temp[p3++] = A[p2++];
        }
    }
    while(p1 <= m){
        temp[p3++] = A[p1++];
    }
    while(p2 <= e){
        temp[p3++] = A[p2++];
    }

    int k = 0;
    for(int i = s; i <= e; i++){
        A[i] = temp[k++];
    }
}
void mergesort(vector<int>& A, int s, int e, int& ans){
    //base case
    if(s == e)return;

    int m = s + (e - s)/2;
    mergesort(A,s,m,ans);
    mergesort(A,m+1,e,ans);
    merge(A,s,m,e,ans);
    
}
int Solution::solve(vector<int> &A) {
    int ans = 0;
    mergesort(A,0,A.size() - 1, ans);
    return ans;
}
