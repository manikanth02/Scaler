vector<int> Solution::plusOne(vector<int> &A) {
    int n = A.size();
    int carry = 1;

    for(int i = n - 1; i >= 0; i--){
        int sum = A[i] + carry;
        A[i] = sum%10;
        carry = sum/10;
    }

    if(carry > 0){
        A.insert(A.begin(),carry);
    }

    while(A.size() > 1 && A[0] == 0){
        A.erase(A.begin());
    }
    return A;
}
