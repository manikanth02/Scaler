idea - 1:: O(rowSize * colSize)



int Solution::solve(vector<vector<int> > &A) {
    int n = A.size();
    int maxm = INT_MIN;
    int ans = -1;

    for(int i = 0;i < n; i++){
        int count = 0;
        for(int j = 0; j < n; j++){
            count += A[i][j];
        }
        A[i][n - 1] = count;
    }

    for(int i = 0; i < n; i++){
        if(A[i][n - 1] > maxm){
            maxm = A[i][n - 1];
            ans = i;
        }
    }
    return ans;
}


idea-2 :: use the fetaures of row is sorted
T.C:: O(rowSize * colSize)  




int Solution::solve(vector<vector<int> > &A) {
    int n = A.size();

    int i = 0;
    int j = n - 1;
    int ans = -1;

    while(i < n && j >= 0){
        if(A[i][j] ==  1){
            j--;
            ans = i;
        }else{
            i++;
        }
    }

    return ans;
}
