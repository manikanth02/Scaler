// Basically this qn is for observation => When we flip 0 -> 1 then we are doing +1 and when 
// we are doing 1 -> 0 then we are doing (-1),and
// we can consider this as kadane's Algorithm

vector<int> Solution::flip(string A) {
int n = A.size();
vector<int>ans(n,0);

for(int i = 0; i < n; i++){
if(A[i] == '0'){
    ans[i] = 1;
}else{
    ans[i] = -1;
}
}

int currentSum = 0;
int maxmSum = 0;
int start = 0;
int currentStart = 0;
int end = 0;

for(int i = 0; i < n; i++){
    currentSum += ans[i];
    if(currentSum > maxmSum){
        maxmSum = currentSum;
        start = currentStart;
        end = i;
    }
    if(currentSum < 0){
        currentSum = 0;
        currentStart = i + 1;
    }
}

if(maxmSum <= 0){
    return {};
}
return {start + 1,end + 1};

}
