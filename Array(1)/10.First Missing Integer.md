int Solution::firstMissingPositive(vector<int> &A) {
    int n = A.size();
    
    // Step 1: Rearrange the array
    for (int i = 0; i < n; ++i) {
        while (A[i] > 0 && A[i] <= n && A[A[i] - 1] != A[i]) {
            swap(A[i], A[A[i] - 1]);
        }
    }
    
    // Step 2: Identify the missing positive integer
    for (int i = 0; i < n; ++i) {
        if (A[i] != i + 1) {
            return i + 1;
        }
    }
    
    // If all elements are in place, return N + 1
    return n + 1;
}
