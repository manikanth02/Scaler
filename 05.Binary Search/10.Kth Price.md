int findMinmCount(vector<int> A, int & mid){
    int n = A.size();
    int minimumNumb = 0;
    for(int i = 0; i < n;i++){
        if(A[i] <= mid){
            minimumNumb++;
        }
    }
    return minimumNumb;
}
int Solution::solve(const vector<int> &A, int k) {

    int start = A[0];
    int end = A[0];
    int n = A.size();

    for(int i = 1; i < n; i++){
        start = min(start,A[i]);
        end = max(end,A[i]);
    }

    int ans = -1;

    while(start <= end){
        int mid = start + (end - start)/2;
        int minimumNumb = findMinmCount(A,mid);
            if(minimumNumb >= k){
                ans = mid;
                end = mid - 1;
            }else{
                start = mid + 1;
            }
    }

    return ans;


}
