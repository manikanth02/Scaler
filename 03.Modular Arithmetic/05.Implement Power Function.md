int Solution::pow(int A, int B, int C) {
    
    
    long long base = A;
    long long result = 1;

    while(B > 0){
        if(B%2 == 1){
            result = (result * base)%C;
        }
        B = B/2;
        base = (base * base)%C;
    }
    return (result + C)%C;
}
