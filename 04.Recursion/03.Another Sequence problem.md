```
int findAthSequenec(int A){
    if(A == 0){
        return 1;
    }
    if(A == 1){
        return 1;
    }
    if(A == 2){
        return 2;
    }
    return findAthSequenec(A - 1) + findAthSequenec(A - 2) + findAthSequenec(A - 3) + A;
}
int Solution::solve(int A) {
    if(A == 0){
        return 1;
    }
    if(A == 1){
        return 1;
    }
    if(A == 2){
        return 2;
    }
    return findAthSequenec(A);
}
```
